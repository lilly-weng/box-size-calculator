
            function productDetails() {
                let table = document.getElementById("details");
                let row = table.insertRow(-1);
                let r = table.getElementsByTagName("tr");

                let number = row.insertCell(0);
                let length = row.insertCell(1);
                let width = row.insertCell(2);
                let height = row.insertCell(3);
                let action = row.insertCell(4);

                //make length, width, height editable
                length.contentEditable = true;
                width.contentEditable = true;
                height.contentEditable = true;

                //prevent enter when editing
                length.addEventListener('keydown', (event) => {
                    if (event.keyCode === 13) {
                        event.preventDefault();
                    }
                });

                width.addEventListener('keydown', (event) => {
                    if (event.keyCode === 13) {
                        event.preventDefault();
                    }
                });

                height.addEventListener('keydown', (event) => {
                    if (event.keyCode === 13) {
                        event.preventDefault();
                    }
                });

                //generate row number
                for (let i = 0, len = r.length; i < len; i++) {
                    number.innerHTML = i
                }

                //catch input value to the table
                length.innerHTML = document.getElementById("length").value;
                width.innerHTML = document.getElementById("width").value;
                height.innerHTML = document.getElementById("height").value;
                //generate delete button
                action.outerHTML = "<tr><td><input type='button' id = 'delete_btn' value='Delete' onclick='deleteValue()' /></td></tr>";

            }

            //delete function
            function deleteValue(row) {

                let tr = td.parentNode;
                tr.parentNode.removeChild(tr);

            }
